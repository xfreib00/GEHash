<s> ::= <lines>

<lines> ::= <line> <lines> | <line>

<line> ::= "hash = " <stmt> ";"

<stmt> ::= <keys> <op> <keys> | "~(" <has> ")" | <has> "<<" <aux>

<op> ::= "+" | "*" | "^"

<aux> ::= "3" | "5" | "7"

<keys> ::= "key[" <number> "]"

<number> ::= "0" | "1" | "2" | "3" | "4" | "5" | "6" | "7" | "8"

<has> ::= "hash"
